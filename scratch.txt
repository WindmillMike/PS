#a.
import numpy as np
import matplotlib.pyplot as plt

frecventa = 5
frecventaEsantionare = 1000
durata = 1
t = np.arange(0, durata, 1 / frecventaEsantionare)
x = np.sin(2 * np.pi * frecventa * t + 0.5 * np.pi)

omega = 1
y = x * np.exp(-2j * np.pi * omega * t)

fig, axes = plt.subplots(1, 2, figsize=(15, 6))

axes[0].plot(t * frecventaEsantionare, x, color='green')
axes[0].set_title("Semnalul x[n] in Domeniul Timp")
axes[0].set_xlabel("Timp (esantioane)")
axes[0].set_ylabel("Amplitudine")
axes[0].axhline(0, color='black', linewidth=0.5)
axes[0].set_xlim([0, 1000])

index_marker = 620
axes[0].plot(index_marker, x[index_marker], 'ro', markersize=8)
axes[0].plot([index_marker, index_marker], [0, x[index_marker]], 'r-', linewidth=1)

axes[1].plot(y.real, y.imag, color='slateblue', linewidth=1.5)
axes[1].set_xlabel("Real")
axes[1].set_ylabel("Imaginar")

axes[1].axhline(0, color='black', linewidth=0.5)
axes[1].axvline(0, color='black', linewidth=0.5)
axes[1].set_aspect('equal', adjustable='box')
axes[1].set_xlim([-1.1, 1.1])
axes[1].set_ylim([-1.1, 1.1])


axes[1].plot(y[index_marker].real, y[index_marker].imag, 'ro', markersize=8)
axes[1].plot([0, y[index_marker].real], [0, y[index_marker].imag], 'r-', linewidth=1)

plt.tight_layout()
plt.show()



#b.
fig, axes = plt.subplots(1, 4, figsize=(15, 6))

def showFig(y, i):
    axes[i].plot(y.real, y.imag, color='pink', linewidth=1.5)
    axes[i].set_xlabel("Real")
    axes[i].set_ylabel("Imaginar")

    axes[i].axhline(0, color='black', linewidth=0.5)
    axes[i].axvline(0, color='black', linewidth=0.5)
    axes[i].set_aspect('equal', adjustable='box')
    axes[i].set_xlim([-1.1, 1.1])
    axes[i].set_ylim([-1.1, 1.1])
    dft_result = np.mean(y)
    axes[i].plot([0, dft_result.real], [0, dft_result.imag], 'k-', linewidth=4)
    axes[i].plot(dft_result.real, dft_result.imag, 'ko', markersize=10)

omega = [1, 2, 5, 7]
for i in range(4):
    y = x * np.exp(-2j * np.pi * omega[i] * t)
    showFig(y, i)

plt.tight_layout()
plt.show()
